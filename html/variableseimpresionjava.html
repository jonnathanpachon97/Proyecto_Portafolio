                        nombre = input.nextLine();

                        apellidos = input.nextLine();
                        System.out.print("Digite el nombre de su madre: ");
                        nombreMadre = input.nextLine();
                        System.out.print("Digite los apellidos de su madre: ");
                        apellidosMadre = input.nextLine();

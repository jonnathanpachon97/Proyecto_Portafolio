                public class Ejercicio7_Condicionales {

                        double peso, estatura, imc;

                        nombre = input.nextLine();

                        edad = Integer.parseInt(input.nextLine());
                        System.out.println("Digite su peso en kilogramos: ");
                        peso = Double.parseDouble(input.nextLine());
                        System.out.println("Digite su estatura en metros: ");

                        imc = peso / (Math.pow(estatura,2));

                        if (imc < 18.5){
                            System.out.println("Su nivel de peso es BAJO");
                        }
                        else if (imc >= 18.5 && imc <= 24.9){
                            System.out.println("Su nivel de peso es NORMAL");

                        }

                            System.out.println("Su nivel de peso es OBESO");
                        }

                }

9. El profesor de geometría está explicando a sus estudiantes las fórmulas para calcular el área de diferentes figuras geométricas, para ello requiere una aplicación que le facilite el ejercicio solicitándole los valores al estudiante. La aplicación debe permitir que el estudiante seleccione si desea calcular el área de un rectángulo, triángulo o trapecio. No olvide solicitar los datos necesarios para realizar cada cálculo y mostrar su respectivo resultado.
                import java.util.Scanner;

                public class Ejercicio9_Condicionales {
                    public static void main(String[] args) {
                        Scanner input = new Scanner(System.in);

                        int opcion;
                        double base, altura, resultado, baseInferior, baseSuperior;

                        System.out.println("---CALCULADORA DE AREAS---");
                        System.out.println("" +
                                "1. RECTANGULO" +
                                "\n2. TRIANGULO" +
                                "\n3. TRAPECIO");
                        opcion = Integer.parseInt(input.nextLine());

                        switch (opcion){
                            case 1:
                                System.out.println("Digite la base en metros: ");
                                base = Double.parseDouble(input.nextLine());
                                System.out.println("Digite la altura en metros: ");
                                altura = Double.parseDouble(input.nextLine());

                                resultado = (base * altura);
                                System.out.println("El are del rectangulo es: "+resultado+"m^2");
                                break;

                            case 2:
                                System.out.println("Digite la base en metros: ");
                                base = Double.parseDouble(input.nextLine());
                                System.out.println("Digite la altura en metros: ");
                                altura = Double.parseDouble(input.nextLine());

                                resultado = (base * altura)/2;
                                System.out.println("El are del triangulo es: "+resultado+"m^2");
                                break;

                            case 3:
                                System.out.println("Digite la base inferior en metros: ");
                                baseInferior = Double.parseDouble(input.nextLine());
                                System.out.println("Digite la base Superior en metros: ");
                                baseSuperior = Double.parseDouble(input.nextLine());
                                System.out.println("Digite la altura en metros: ");
                                altura = Double.parseDouble(input.nextLine());

                                resultado = ((baseSuperior + baseInferior)*altura)/2;
                                System.out.println("El area del trapecio es: "+resultado+"m^2");
                                break;

                            default:
                                System.out.println("Opcion invalida!!! Intente de nuevo");

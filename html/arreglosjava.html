3. Imprimir los números primos del 1 al 1000, el resultado debe ser buscado de forma matemática.
                public class Ejercicio3_Arreglos {
                    public static void main(String[] args) {
                
                        int contador;
                        int j, i;
                
                        j = 0;
                
                        int numerosPrimos[] = new int[168];
                
                        for (int numeros1=1;numeros1<=1000;numeros1++){
                            i = 1;
                            contador = 0;
                            while (i <=numeros1){
                                if (numeros1 % i == 0){
                                    contador++;
                                }
                                i++;
                            }
                            if (contador == 2){
                                numerosPrimos[j] = numeros1;
                                j++;
                            }
                        }
                        System.out.print("Numeros primos del 1 al 1000 son: ");
                        for (int numeros1=0;numeros1<=(168-1);numeros1++){
                            System.out.print(numerosPrimos[numeros1]+",");
                        }
                    }
                }

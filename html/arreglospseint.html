Ejercicio 1

1. Crear un vector de tipo Entero con 5 posiciones, llenarlo con información solicitada al usuario. Después de recoger toda información, se requiere imprimir el índice de cada posición en el arreglo con su valor de la siguiente manera: [0] = 55 [1] = 99 [2] = 11 [3] = 56 [4] = 69
                Proceso Arreglos_Ejercicio1
	
                    Definir i Como Entero;
                    Definir numerosEnteros Como Entero;
                    
                    Dimension numerosEnteros[5];
                    
                    
                    Para i <- 0 Hasta 4 Con Paso 1 Hacer
                        Escribir Sin Saltar "Digite el numero entero de la posicion", " ",i,":";
                        Leer numerosEnteros[i];
                    FinPara
                    
                    para i <- 0 Hasta 4 Con Paso 1 Hacer
                        Escribir " [",i,"] ",numerosEnteros[i];
                    FinPara
                    
                FinProceso

Ejercicio 2

2. Crear un arreglo de números enteros de 20 posiciones, el cual, debe ser llenado con números aleatorios entre 1 y 100; despué de haber llenado dicho arreglo, se debe volver a recorrer utilizando un ciclo diferente al que se usó para llenarse e imprimir los números pares e impares. Ejemplo
Números pares: 2, 4, 6, 8, 10
Números impares: 1, 3, 5, 7, 9
                Proceso Arreglos_Ejercicio2
	
                    Definir i, j, k, l Como Entero;
                    Definir numerosEnteros, pares, impares, conteoPares, conteoImpares Como Entero;
                    
                    i <- 0;
                    j <- 0;
                    k <- 0;
                    l <- 0;
                    conteoPares <- 0;
                    conteoImpares <- 0;
                    
                    Dimension numerosEnteros[20];
                    Dimension pares[20];
                    Dimension impares[20];
                    
                    para i <- 0 Hasta 19 Con Paso 1 Hacer
                        numerosEnteros[i] <- Aleatorio(1,100);
                        Escribir " [",i,"] ",numerosEnteros[i];
                        
                    FinPara
                    
                    Mientras j <= 19 Hacer
                        si numerosEnteros[j] % 2 = 0 Entonces
                            pares[conteoPares] <- numerosEnteros[j];
                            conteoPares <- conteoPares + 1;
                        SiNo
                            impares[conteoImpares] <- numerosEnteros[j];
                            conteoImpares <- conteoImpares + 1;
                        FinSi
                        j <- j + 1;
                    FinMientras
                    
                    Escribir " ";
                    Escribir Sin Saltar "Numeros pares: ";
                    
                    Para k <- 0 Hasta (conteoPares - 1) Con Paso 1 Hacer
                        Escribir sin saltar pares[k],",";
                    FinPara
                    
                    Escribir " ";
                    Escribir Sin Saltar "Numero impares: ";
                    
                    Para l <- 0 Hasta (conteoImpares - 1) Con Paso 1 Hacer
                        Escribir sin saltar impares[l],",";
                    FinPara
                    
                    Escribir " ";
                    
                FinProceso

Ejercicio 3

3. Imprimir los números primos del 1 al 1000, el resultado debe ser buscado de forma matemática.
                Proceso Arreglos_Ejercicio3
	
                    Definir i, j, num, contador,numerosPrimos Como Entero;
                    Dimension numerosPrimos[168];
                    
                    j <- 0;
                    
                    Para num <- 1 hasta 1000 Con Paso 1 Hacer
                        i <- 1;
                        contador <- 0;
                        Mientras i <= num Hacer
                            Si num % i = 0 Entonces
                                contador <- contador + 1;
                            FinSi
                            i <- i + 1;
                        FinMientras
                        Si contador = 2 Entonces
                            numerosPrimos[j] <- num;
                            j <- j + 1;
                        FinSi
                    FinPara
                    
                    Escribir "Numeros Primos del 1 a 1000 son:";
                    
                    Para num <- 0 Hasta (168 - 1) Con Paso 1 Hacer
                        Escribir Sin Saltar numerosPrimos[num],",";
                    FinPara
                    Escribir " ";

                FinProceso
